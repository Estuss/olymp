from functools import lru_cache, cache

@lru_cache(maxsize=None)                  # кэш на всё
def fib(n):
    if n < 2: return n
    return fib(n-1) + fib(n-2)            # O(n) вместо O(2^n)

@cache                                    # Python 3.9+ (lru_cache без аргументов)
def factorial(n):
    return 1 if n <= 1 else n * factorial(n-1)

# Очистка кэша
